  // ‚úÖ SDK v3.x: –ü—Ä–∞–≤–∏–ª—å–Ω–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ useLaunchParams

    const encodedTitle = encodeURIComponent(entity.name || entity.title);
